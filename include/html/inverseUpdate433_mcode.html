1function [ inv_A_update ] = inverseUpdate(inv_A,b,c)
 2%% INVERSE UPDATE USING BLOCK INVERSE LEMMA
 3%inv_A_update = inverseUpdate(inv_A,b,c) computes the inverse of a matrix obtained
 4%formula : inv([A b;b' c]).  b is column vectors and c is a real number.
 5%input  : The Inverse of Matrix A and vectors b and real number c  -->  (inv_A,b,c)
 6%output : The Inverse Matrix of [A b;b' c]  -->  inv_A_update
 7    i_k = 1/(c - b'*inv_A*b);
 8    inv_A_update = [inv_A+i_k*inv_A*b*b'*inv_A -i_k*inv_A*b;-i_k*b'*inv_A i_k];
 9end
